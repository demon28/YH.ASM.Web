-- Create table
create table TASM_CUSTOMER
(
  cid        NUMBER(9) not null,
  name       VARCHAR2(100) not null,
  phone      VARCHAR2(100),
  emial      VARCHAR2(100),
  adress     VARCHAR2(200),
  status     NUMBER(1) default 0 not null,
  isdel      NUMBER(1) default 0,
  createtime DATE default sysdate,
  remarks    VARCHAR2(200)
)
;
